Model Architecture

In[38]:
# Building the sequential CNN model
model = Sequential()

# First convolutional layer
model.add(Conv2D(32, (3, 3), activation='relu', input_shape=(img_size, img_size, 1)))
model.add(ZeroPadding2D(padding=(1, 1)))
model.add(MaxPooling2D(pool_size=(2, 2)))

# Second convolutional layer
model.add(Conv2D(64, (3, 3), activation='relu'))
model.add(ZeroPadding2D(padding=(1, 1)))
model.add(MaxPooling2D(pool_size=(2, 2)))

# Third convolutional layer
model.add(Conv2D(128, (3, 3), activation='relu'))
model.add(ZeroPadding2D(padding=(1, 1)))
model.add(MaxPooling2D(pool_size=(2, 2)))

# Flatten layer
model.add(Flatten())

# Fully connected layers
model.add(Dense(128, activation='relu'))
model.add(Dropout(0.1))  # Added dropout for regularization
model.add(Dense(num_classes+1, activation='softmax'))  # Use softmax for multi-class classification

# Displaying model summary
model.summary()

Experimenting with different Learning Rates

In[43]:
# Defining different learning rates to experiment on
learning_rates = [0.01, 0.001, 0.0001]
histories = []
models = []


 # Using loop for Training the model with each learning rate
for lr in learning_rates:
    print(f"\nTraining model with learning rate: {lr}")

    # Creating a copy of the model to train with this learning rate so there is no change in previous code
    model_copy = tf.keras.models.clone_model(model)

    # Compiling the model with the defined learning rate
    model_copy.compile(optimizer=Adam(learning_rate=lr),
                      loss='categorical_crossentropy',
                      metrics=['accuracy'])

    # Training the model
    history = model_copy.fit(
        X_train, y_train,
        epochs=5,
        batch_size=32,
        validation_data=(X_val, y_val),
        verbose=1
    )

    # Saving the history and model
    histories.append(history)
    models.append(model_copy)
